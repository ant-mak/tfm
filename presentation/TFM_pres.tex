\section{Generación de arte}
\subsection{DCGAN}
%------------------------------------------------

\begin{frame}
\frametitle{DCGAN}
\center \textbf{Deep Convolutional Generative Adversarial Network}
\vfill
\begin{itemize}
	\item Discriminador: Convolución con stride
	\item Generador: Convolución fraccional con stride
	\item No utilizar capas totalmente conectadas
	\item Generador: ReLU $+ \tanh$
	\item Discriminador: LeakyReLU $+$ sigmoide
	\item BatchNorm
	\item Inicialización gaussiana
	\item <2-> One sided label smoothing
\end{itemize}
\end{frame}
